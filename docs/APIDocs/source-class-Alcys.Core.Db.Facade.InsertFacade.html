 26 use Alcys\Core\Db\Factory\DbFactoryInterface;

 30 /**
 31  * Class InsertFacade
 32  * @package Alcys\Core\Db\Facade
 33  */
 34 class InsertFacade implements InsertFacadeInterface
 35 {
 36     /**
 37      * @var \PDO
 38      */
 39     private $pdo;
 40 
 41     /**
 42      * @var InsertInterface
 43      */
 44     private $insert;
 45 
 46     /**
 47      * @var DbFactoryInterface
 48      */
 49     private $factory;
 50 
 51 
 52     /**
 53      * Initialize the insert facade.
 54      * Set all required properties to update entries in the database.
 55      *
 56      * @param \PDO               $pdo     PDO connection.
 57      * @param InsertInterface    $insert  Insert statement.
 58      * @param DbFactoryInterface $factory Factory to create several objects.
 59      * @param string             $table   Name of the table from which should select (tables can get extended).
 60      */
 61     public function __construct(\PDO $pdo, InsertInterface $insert, DbFactoryInterface $factory, $table)
 62     {
 63         $this->pdo     = $pdo;
 64         $this->insert  = $insert;
 65         $this->factory = $factory;
 66 
 67         /** @var TableInterface $tableObj */
 68         $tableObj = $this->factory->references('Table', $table);
 69         $this->insert->table($tableObj);
 70     }
 71 
 72 
 73     /**
 74      * Execute the query and insert the expected entries in the database.
 75      *
 76      * @throws \Exception When an error is occurred while the insertion.
 77      */
 78     public function execute()
 79     {
 80         $query  = $this->factory->builder($this->insert)->process();
 81         $result = $this->pdo->query($query);
 82         if(!$result)
 83         {
 84             throw new \Exception('An error is while the insertion occurred!');
 85         }
 86     }
 87 
 88 
 89     /**
 90      * Set columns to the query in which should insert new entries.
 91      * Afterwards, the value method should call to set the values.
 92      *
 93      * @param array $columns An usual array with the column names as values.
 94      *
 95      * @return $this The same instance to concatenate methods.
 96      */
 97     public function columns(array $columns)
 98     {
 99         $columnsArray = array();
100         foreach($columns as $column)
101         {
102             $columnsArray[] = $this->factory->references('Column', $column);
103         }
104         $this->insert->columns($columnsArray);
105 
106         return $this;
107     }
108 
109 
110     /**
111      * Set values to the query.
112      * The method can execute multiple times. The passed array require the
113      * same amount of elements, otherwise an exception will thrown.
114      *
115      * @param array $values An usual array with the values that should insert.
116      *
117      * @return $this The same instance to concatenate methods.
118      */
119     public function values(array $values)
120     {
121         $valuesArray = array();
122         foreach($values as $value)
123         {
124             $valuesArray[] = $this->factory->references('Value', $value);
125         }
126         $this->insert->values($valuesArray);
127 
128         return $this;
129     }
130 }
