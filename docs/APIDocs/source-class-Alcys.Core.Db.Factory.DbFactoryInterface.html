 40 /**
 41  * Interface DbFactoryInterface
 42  * @package Alcys\Core\Db\Factory
 43  */
 44 interface DbFactoryInterface
 45 {
 46     /**
 47      * Create instances of the type expression.
 48      *
 49      * @param string $name The name of the expression.
 50      *
 51      * @return ExpressionInterface An instance of a condition expression..
 52      * @throws \Exception
 53      */
 54     public function expression($name);
 55 
 56 
 57     /**
 58      * Create an instance of the expected statement type.
 59      *
 60      * @param string $name The name of the statement type.
 61      *
 62      * @return StatementInterface A statement entity.
 63      * @throws \Exception
 64      */
 65     public function statement($name);
 66 
 67 
 68     /**
 69      * Create instances of the type references.
 70      *
 71      * @param string $name The name of the expression.
 72      *
 73      * @return ReferencesInterface An instance of the expected reference type which implements a ReferencesInterface.
 74      * @throws \Exception
 75      */
 76     public function references($name);
 77 
 78 
 79     /**
 80      * Create an instance of a query builder.
 81      *
 82      * @param StatementInterface $statement The statement value object, from which should the a query build.
 83      *
 84      * @return DeleteBuilder|InsertBuilder|SelectBuilder|UpdateBuilder    An instance of QueryBuilder,
 85      *                                                                    switched by the passed argument.
 86      */
 87     public function builder(StatementInterface $statement);
 88 
 89 
 90     /**
 91      * Create instance of facade objects.
 92      * There are four types of facade objects that could created:
 93      * 1.SelectFacade, 2. UpdateFacade, 3. InsertFacade and 4. DeleteFacade.
 94      * Usage: $factory->facade('select', $pdo, $tableName); or similar variants.
 95      *
 96      * @param string      $name       Whether 'select', 'update', 'insert' or 'delete'.
 97      * @param \PDO        $connection An instance of a PDO connection.
 98      * @param string      $tableName  Name of the 'main'-table for the query.
 99      * @param string|null $tableAlias (Optional) If exists, an alias name can also set.
100      *
101      * @return DeleteFacadeInterface|InsertFacadeInterface|SelectFacadeInterface|UpdateFacadeInterface
102      * @throws \Exception
103      */
104     public function facade($name, \PDO $connection, $tableName, $tableAlias = null);
105 
106 
107     /**
108      * Create instance of expression facade objects.
109      * The difference between this and the facade method
110      * is, that the other one is for statements and this
111      * method for expression facade objects.
112      *
113      * @param string $name Name of the facade object.
114      *
115      * @return ConditionFacade
116      * @throws \Exception
117      */
118     public function expressionFacade($name);
