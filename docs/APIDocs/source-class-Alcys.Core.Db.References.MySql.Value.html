 30 /**
 31  * Class Value
 32  * @package Alcys\Core\Db\References
 33  */
 34 class Value implements ReferencesInterface, ValueInterface, WhereCompareInterface, WhereNumericCompareInterface
 35 {
 36     /**
 37      * Name of the value.
 38      *
 39      * @var string The property that represent the value name.
 40      */
 41     private $value;
 42 
 43 
 44     /**
 45      * Initialize the object that represent a MySql Value.
 46      * The constructor validate the value and throw, if necessary,
 47      * an exception.
 48      *
 49      * @param string $value
 50      */
 51     public function __construct($value)
 52     {
 53         if(is_string($value))
 54         {
 55             $this->value = '"' . htmlentities($value, ENT_QUOTES) . '"';
 56         }
 57         elseif($value instanceof Numeric || is_numeric($value))
 58         {
 59             $this->value = (string)htmlentities($value, ENT_QUOTES);
 60         }
 61         elseif($value instanceof Column)
 62         {
 63             $this->value = $value->getColumnName();
 64         }
 65         else
 66         {
 67             throw new \InvalidArgumentException('passed argument has an illegal type');
 68         }
 69     }
 70 
 71 
 72     /**
 73      * Return the value that should used from the where expression
 74      * object to compare with an other column.
 75      *
 76      * @return string
 77      */
 78     public function getCompareValue()
 79     {
 80         return $this->getValue();
 81     }
 82 
 83 
 84     /**
 85      * Return the value that should used from the where expression
 86      * object to compare with other numeric values.
 87      *
 88      * @return string
 89      */
 90     public function getNumericCompareValue()
 91     {
 92         return $this->getValue();
 93     }
 94 
 95 
 96     /**
 97      * @return string
 98      */
 99     public function getValue()
100     {
101         return $this->value;
102     }
103 }
