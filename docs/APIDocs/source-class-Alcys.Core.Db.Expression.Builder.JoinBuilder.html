 74     /**
 75      * Build the join expression string.
 76      * Processing the passed join expression and return it in form of a string.
 77      *
 78      * @see JoinBuilder::_checkJoinArray
 79      * @see JoinBuilder::_prepareJoinTable
 80      * @see JoinBuilder::_prepareJoinCondition
 81      *
 82      * @return string The join expression as a string.
 83      * @throws \Exception When the join expression has an invalid format.
 84      */
 85     public function build()
 86     {
 87         $this->_checkJoinArray();
 88         $this->way = ($this->joinArray['way']) ?
 89             strtoupper(str_replace(':', ' ', $this->joinArray['way'])) . ' ' : null;
 90         $this->_prepareJoinTable($this->joinArray['tables']);
 91         $this->conditionType = strtoupper($this->joinArray['conditionType']) . ' ';
 92 
 93         if($this->joinArray['conditionType'] !== 'natural')
 94         {
 95             $this->_prepareJoinCondition($this->joinArray['condition']);
 96             $expression = $this->way . 'JOIN' . $this->tables . $this->conditionType . $this->condition;
 97         }
 98         else
 99         {
100             $expression = $this->conditionType . $this->way . 'JOIN' . rtrim($this->tables);
101         }
102 
103         return $expression . ' ';
104     }

130     /**
131      * Compare the join array with the expressions default join array and throw an exception if they are the same.
132      *
133      * @throws \Exception
134      */
135     private function _checkJoinArray()
136     {
137         $defaultJoinArray = array('way' => null, 'tables' => null, 'conditionType' => null, 'condition' => null);
138         if($this->joinArray === $defaultJoinArray)
139         {
140             throw new \Exception('one of the methods: inner(), natural(), left[Outer](), right[Outer] must called before!');
141         }
142     }

145     /**
146      * Validate the tables array to a string.
147      *
148      * @param TableInterface[] $tables
149      */
150     private function _prepareJoinTable(array $tables)
151     {
152         $this->tables = ' ';
153         foreach($tables as $tableObj)
154         {
155             $this->tables .= $tableObj->getExpression() . ', ';
156         }
157         $this->tables = rtrim(rtrim($this->tables), ',') . ' ';
158     }

161     /**
162      * Validate the columns array to a string.
163      *
164      * @param ColumnInterface[] $condition
165      *
166      * @return string
167      */
168     private function _prepareJoinCondition(array $condition)
169     {
170         if(count($condition) === 1)
171         {
172             return $this->condition = '(' . $condition[0]->getColumnName() . ')';
173         }
174 
175         return $this->condition =
176             $condition[0]->getTableName() . '.' . $condition[0]->getColumnName() . ' = ' . $condition[1]->getTableName()
177             . '.' . $condition[1]->getColumnName();
178     }
