237     /**
238      * Add a like condition to the condition array.
239      * The order of the method call is important, in which way the condition string will
240      * concatenate.
241      *
242      * @param ColumnInterface     $column Name of the column.
243      * @param ReferencesInterface $value  Value for the expression.
244      * @param int                 $level  The several levels of a like expression, see private likeLevel - methods.
245      *
246      * @see Alcys\Core\Db\Condition\Condition::_likeLevelOne
247      * @see Alcys\Core\Db\Condition\Condition::_likeLevelTwo
248      * @see Alcys\Core\Db\Condition\Condition::_likeLevelThree
249      *
250      * @return $this The current condition instance to concatenate method invokes.
251      * @throws \Exception
252      */
253     public function like(ColumnInterface $column, ReferencesInterface $value, $level = 0)
254     {
255         if($level === 0)
256         {
257             $condition = $column->getColumnName() . ' LIKE ' . $value->getValue();
258         }
259         elseif($level === 1)
260         {
261             $condition = $column->getColumnName() . ' LIKE ' . $this->_likeLevelOne($value);
262         }
263         elseif($level === 2)
264         {
265             $condition = $column->getColumnName() . ' LIKE ' . $this->_likeLevelTwo($value);
266         }
267         elseif($level === 3)
268         {
269             $condition = $column->getColumnName() . ' LIKE ' . $this->_likeLevelThree($value);
270         }
271         else
272         {
273             throw new \Exception('invalid level argument');
274         }
275         $this->_addCondition($condition);
276 
277         return $this;
278     }

347     /**
348      * This method fill the condition array in the expected format.
349      *
350      * If no logic operator method was invoked after a condition method, it will
351      * throw an exception. Otherwise, after the first call, a condition without
352      * an logic operator will add to the condition array and at the others calls,
353      * a condition with an logic operator will add to the condition array.
354      *
355      * @param string $condition
356      *
357      * @throws \Exception
358      */
359     private function _addCondition($condition)
360     {
361         if($this->logicOperator === null && $this->invoked === false)
362         {
363             $this->conditionArray[] = array($condition);
364             $this->invoked          = true;
365         }
366         elseif($this->logicOperator === null)
367         {
368             throw new \Exception('a logic operator method have to be invoked to continue add conditions');
369         }
370         else
371         {
372             $this->conditionArray[] = array($this->logicOperator => $condition);
373             $this->logicOperator    = null;
374         }
375     }

378     /**
379      * Validate the value and the percentage sign to a valid value for an expression.
380      * The sign will passed at the begin of the value.
381      *
382      * Example:    Value: 123 will validated to: %123.
383      *             Value: "a" will validated to: "%a".
384      *
385      * @param ReferencesInterface $value The value for the expression.
386      *
387      * @return string
388      */
389     private function _likeLevelOne(ReferencesInterface $value)
390     {
391         $regex = '/"/';
392         if(preg_match($regex, $value->getValue()))
393         {
394             return preg_replace($regex, '"%', $value->getValue(), 1);
395         }
396 
397         return '%' . $value->getValue();
398     }

401     /**
402      * Validate the value and the percentage sign to a valid value for an expression.
403      * The sign will passed at the end of the value.
404      *
405      * Example:    Value: 123 will validated to: 123%.
406      *             Value: "a" will validated to: "a%".
407      *
408      * @param ReferencesInterface $value The value for the expression.
409      *
410      * @return string
411      */
412     private function _likeLevelTwo(ReferencesInterface $value)
413     {
414         $regex   = '/("[a-zA-Z0-9_-]*)"/';
415         $matches = array();
416         if(preg_match($regex, $value->getValue(), $matches))
417         {
418             return $matches[1] . '%"';
419         }
420 
421         return $value->getValue() . '%';
422     }
