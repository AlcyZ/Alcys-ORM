165     /**
166      * This method validate and prepare the column name,
167      * or throw an exception if an invalid argument is passed
168      * as parameter.
169      *
170      * @param string $column
171      */
172     private function _prepareColumnArgument($column)
173     {
174         if(!is_string($column) && !$column instanceof SqlFunctionInterface)
175         {
176             throw new \InvalidArgumentException('Argument 1 passed is not the expected type!');
177         }
178         if($column instanceof SqlFunctionInterface || $column === '*')
179         {
180             $this->columnName = (string)htmlentities($column, ENT_QUOTES);
181         }
182         else
183         {
184             $this->columnName = '`' . htmlentities($column, ENT_QUOTES) . '`';
185         }
186     }
